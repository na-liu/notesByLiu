x1_left = -3.0  # x1ï¼Œx2å˜é‡çš„èŒƒå›´
x1_right = 12.1
x2_left = 4.1
x2_right = 5.8
precision = 10 ** -4  # ç²¾åº¦è¦æ±‚
pc = 0.6  # äº¤å‰æ¦‚ç‡

def count_code_gene():
    x1_len = math.ceil(math.log((x1_right - x1_left) / precision + 1, 2))  # 18
    x2_len = math.ceil(math.log((x2_right - x2_left) / precision + 1, 2))  # 15
    # print(x1_len, x2_len)
    gene_length = x1_len + x2_len
    return x1_len, x2_len, gene_length


# ---------ç”Ÿæˆéšæœºä¸ªä½“ï¼Œç¼–ç [input:ç¼–ç ä½æ•°ï¼Œoutput:ä¸ªä½“gene]--------
def create_random_individual():
    gene_str = ''
    for i in range(gene_length):
        gene_str += str(random.randint(0, 1))
    # print(isinstance(gene_str,str))
    return gene_str


# ------åˆå§‹åŒ–ç§ç¾¤;input:ç§ç¾¤è§„æ¨¡ï¼Œgeneä¸ªæ•°ï¼Œoutput:åˆå§‹ç§ç¾¤-----------
def init_generation(population_size):

# ----ä¸ªä½“è§£ç ,input:ä¸ªä½“å’Œä¸¤å˜é‡é•¿åº¦ï¼Œoutput:è§£ç åçš„ä¸¤ä¸ªå˜é‡å€¼--------
def decode(gene_str):
    x1 = int(gene_str[:x1_len], 2)
    x2 = int(gene_str[x1_len:], 2)
    x1 = x1_left + (x1_right - x1_left) * x1 / (2 ** x1_len - 1)
    x2 = x2_left + (x2_right - x2_left) * x2 / (2 ** x2_len - 1)
    # print('è§£ç ï¼š', x1, x2)
    return x1, x2


# --------------------------ç›®æ ‡å‡½æ•°------------------------------------
def fitness(x1, x2):
    return 21.5 + x1 * math.sin(4 * math.pi * x1) + x2 * math.sin(20 * math.pi * x2)


# --------è®¡ç®—ä¸ªä½“é€‚åº”åº¦å€¼;[input:ç§ç¾¤ï¼Œoutput:é€‚åº”åº¦list]--------------
def adaption(generation, count):
    # x1_len, x2_len = count_code_gene()[:2]
    individual = []  # x1,x2
    fit = []  # ä¸ªä½“çš„é€‚åº”åº¦å€¼
    proportion = []  # ä¸ªä½“é€‚åº”åº¦å€¼å æ¯”
    for i in range(count):
        # print('ä¸ªä½“ï¼š ', generation[i])
        x1, x2 = decode(generation[i])
        individual.append([x1, x2])
        fit.append(fitness(x1, x2))
    for i in range(count):
        proportion.append(fit[i] / sum(fit))
    return fit, proportion, individual


# -------è½®ç›˜èµŒé€‰æ‹©mä¸ªç®—æ³•[input:proportionï¼Œoutput:æ‰€é€‰æ‹©çš„ä¸ªä½“]-------
def roulette(proportion, count):

    select = []  # æœ€ç»ˆé€‰å‡ºæ¥çš„ä¸ªä½“ç¼–å·
    for i in range(len(proportion)):  # è®¡ç®—æ¦‚ç‡ç´¯è®¡å€¼ä¸ªæ•°æ­¤å¤„ç”¨äº†countæ‰€ä»¥é”™äº†
        q.append(proportion[i] + sum(proportion[:i]))

    for i in range(count):  # é€‰countä¸ª
        rand = random.uniform(0, 1)  # ç”Ÿæˆ1ä¸ª[0,1]éšæœºæ•°
        for j in range(len(q)):
            if rand <= q[j]:
                select.append(j)
                break
    # print("è½®ç›˜èµŒé€‰æ‹©çš„ç»“æœï¼š", select)
    return select


# ä¸€æ¬¡äº¤å‰[input:çˆ¶ä»£ä¸¤ä¸ªgene_strï¼Œoutput:å­ä»£æ–°çš„ä¸¤ä¸ªgene_str]
def crossover(father, mother):
    a = random.randint(0, gene_length - 1)  # éšæœºæ‰¾ä¸¤ä¸ªäº¤å‰ç‚¹
    b = random.randint(0, gene_length - 1)
    first = min(a, b)
    second = max(a, b)
    # print('äº¤å‰ç‚¹ï¼š', first, second)
    ch1 = father[:first] + mother[first:second] + father[second:]
    ch2 = mother[:first] + father[first:second] + mother[second:]
    return ch1, ch2

def mutation(individual):

    # é€‰æ‹©å‡ºæ¥çš„Nä¸ªçˆ¶ä»£ä¸ªä½“è¿›è¡ŒN/2æ¬¡äº¤å‰äº§ç”Ÿçš„Nä¸ªå­ä»£ä¸ªä½“ï¼Œåœ¨è¿›è¡ŒNæ¬¡å˜å¼‚
